CREATE OR REPLACE bloque_anidado as
mensaje1 VARCHAR2(100) := 'Hola';
mensaje2 VARCHAR2(100) := mensaje1 || ' Mundo';     
    BEGIN
       DBMS_OUTPUT.PUT_LINE(mensaje2);
       EXCEPTION
       WHEN OTHERS
        THEN
          DBMS_OUTPUT.PUT_LINE(dbms_utility.format_error_stack);
END;
END bloque_anidado;
